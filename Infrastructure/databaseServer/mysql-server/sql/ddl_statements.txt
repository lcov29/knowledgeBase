---------- SHOW ENTITIES -----------------------
show databases;					// show databases
show tables;					// show tables and views
describe <table_name>				// show columns of specific table

show full tables
where table_type = 'VIEW';			// show all views

show procedure status;				// show all procedures

show create table <table_name>;			// show definition of specific table
show create view <view_name>;			// show definition of specific view
show create procedure <procedure_name>;		// show definition of specific procedure

---------- CREATE ENTITIES -----------------------
create [or replace] view <view_name> as <select_statement>;

---------- Alter ENTITIES -----------------------
alter view <view_name> as <select_statement>;
